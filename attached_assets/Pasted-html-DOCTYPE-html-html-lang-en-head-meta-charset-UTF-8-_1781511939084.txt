        // Live sort list options
        function sortPoems() {
            const method = document.getElementById('sort-select').value;
            if (method === 'latest') {
                state.poems.sort((a, b) => new Date(b.date) - new Date(a.date));
            } else if (method === 'popular') {
                state.poems.sort((a, b) => parseFloat(calculateAverageRating(b.ratings)) - parseFloat(calculateAverageRating(a.ratings)));
            } else if (method === 'alphabetical') {
                state.poems.sort((a, b) => a.author.localeCompare(b.author));
            }
            renderPoems();
        }

            event.preventDefault();

            const country = document.getElementById('sub-country').value.trim();
            const title = document.getElementById('sub-title').value.trim();
            const content = document.getElementById('sub-content').value.trim();

        // Approve Submission Action
        function approveSubmission(subId) {
            const index = state.submissions.findIndex(s => s.id === subId);
            if (index !== -1) {
                const sub = state.submissions[index];
                
                state.poems.unshift({
                    id: 'poem_' + Date.now(),
                    title: sub.title,
                    author: sub.author,
                    country: sub.country,
                    content: sub.content,
                    date: sub.date,
                    ratings: [5],
                    reviews: [],
                    category: 'Loudthotz Selection'
                });

                state.submissions.splice(index, 1);
                
                showToast("Piece Approved!", `"${sub.title}" is now available in the Curated Gallery.`);
                updatePendingBadges();
                renderAdminQueue();
                renderPoems();
            }
        }

        // Discard draft action
        function declineSubmission(subId) {
            const index = state.submissions.findIndex(s => s.id === subId);
            if (index !== -1) {
                state.submissions.splice(index, 1);
                showToast("Submission Discarded", "The draft was removed from the review console.");
                updatePendingBadges();
                renderAdminQueue();
            }
        }

        // Update pending items notification counters
        function updatePendingBadges() {
            const count = state.submissions.length;
            const badge = document.getElementById('nav-pending-badge');
            badge.innerText = count;

            } else {
                badge.className = "bg-amber-500 text-black px-1.5 py-0.5 rounded-full text-[10px] font-bold";
            }
        }

        // Publish review action
        function handleReviewSubmit(event) {
            event.preventDefault();
            const username = document.getElementById('rev-username').value.trim();
            const stars = parseInt(document.getElementById('rev-stars').value);
            const text = document.getElementById('rev-text').value.trim();

            const poem = state.poems.find(p => p.id === activePoemDetailId);
            if (poem) {
                poem.reviews.unshift({
                    user: username,
                    rating: stars,
                    text: text
                });
                poem.ratings.push(stars);
                showToast("Critique Published", "Your review has been cataloged under the community panels.");
                openPoemDetailModal(activePoemDetailId);
                renderPoems();
            }
        }

        // Star rating click action
        function ratePoem(score) {
            const poem = state.poems.find(p => p.id === activePoemDetailId);
            if (poem) {
                poem.ratings.push(score);
                showToast("Rating Lodged", `You registered a ${score} star score for "${poem.title}".`);
                openPoemDetailModal(activePoemDetailId);
                renderPoems();
            }
        }
